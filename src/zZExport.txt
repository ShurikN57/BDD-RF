Attribute VB_Name = "zZExport"
Option Explicit

Private Const NOM_THISWORKBOOK As String = "zRFThisWorkbook"
Private Const NOM_FEUILLE_CIBLE As String = "GMC"
Private Const DOSSIER_EXPORT As String = "C:\Users\FMF00CDN\Desktop\BDD-RF-GitHub\src"

Public Sub ExporterToutEnTXT()

    Dim vbComp As Object
    Dim dossier As String
    Dim fso As Object
    Dim nomFichier As String
    Dim codeTexte As String
    Dim cheminTempFrm As String
    Dim nomComp As String
    
    dossier = DOSSIER_EXPORT
    
    Set fso = CreateObject("Scripting.FileSystemObject")
    If Not fso.FolderExists(dossier) Then
        fso.CreateFolder dossier
    End If
    
    For Each vbComp In ThisWorkbook.VBProject.VBComponents
        
        nomComp = vbComp.Name
        
        Select Case vbComp.Type
        
            Case 1, 2   ' Module standard / Classe
                nomFichier = dossier & "\" & nomComp & ".txt"
                EcrireTexteDansFichierUTF8 nomFichier, LireCodeComponent(vbComp, nomComp)
            
            Case 3      ' UserForm
                cheminTempFrm = dossier & "\" & nomComp & ".frm"
                
                On Error Resume Next
                Kill cheminTempFrm
                Kill dossier & "\" & nomComp & ".frx"
                Kill dossier & "\" & nomComp & ".txt"
                On Error GoTo 0
                
                vbComp.Export cheminTempFrm
                
                nomFichier = dossier & "\" & nomComp & ".txt"
                codeTexte = LireFichierTexteSysteme(cheminTempFrm)
                EcrireTexteDansFichierUTF8 nomFichier, codeTexte
                
                On Error Resume Next
                Kill cheminTempFrm
                Kill dossier & "\" & nomComp & ".frx"
                On Error GoTo 0
            
            Case 100    ' ThisWorkbook / Feuilles
                If nomComp = NOM_THISWORKBOOK Or nomComp = NOM_FEUILLE_CIBLE Then
                    nomFichier = dossier & "\" & nomComp & ".txt"
                    EcrireTexteDansFichierUTF8 nomFichier, LireCodeComponent(vbComp, nomComp)
                End If
        
        End Select
    Next vbComp
    
    MsgBox "Export TXT UTF-8 terminé dans : " & dossier, vbInformation

End Sub

Private Function LireCodeComponent(ByVal vbComp As Object, ByVal nomComp As String) As String

    Dim codeTexte As String
    
    If vbComp.CodeModule.CountOfLines > 0 Then
        codeTexte = vbComp.CodeModule.Lines(1, vbComp.CodeModule.CountOfLines)
    Else
        codeTexte = ""
    End If
    
    LireCodeComponent = "Attribute VB_Name = """ & nomComp & """" & vbCrLf & codeTexte

End Function

Private Sub EcrireTexteDansFichierUTF8(ByVal chemin As String, ByVal contenu As String)

    Dim stm As Object
    
    On Error Resume Next
    Kill chemin
    On Error GoTo 0
    
    Set stm = CreateObject("ADODB.Stream")
    
    With stm
        .Type = 2
        .Charset = "utf-8"
        .Open
        .WriteText contenu
        .SaveToFile chemin, 2
        .Close
    End With
    
    Set stm = Nothing

End Sub

Private Function LireFichierTexteSysteme(ByVal chemin As String) As String

    Dim fso As Object
    Dim ts As Object
    
    Set fso = CreateObject("Scripting.FileSystemObject")
    Set ts = fso.OpenTextFile(chemin, 1, False, -2)
    
    LireFichierTexteSysteme = ts.ReadAll
    
    ts.Close
    Set ts = Nothing
    Set fso = Nothing

End Function
