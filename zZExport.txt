Attribute VB_Name = "zZExport"
Option Explicit

Private Const NOM_THISWORKBOOK As String = "zRFThisWorkbook"
Private Const NOM_FEUILLE_CIBLE As String = "GMC"

Public Sub ExporterToutEnTXT()

    Dim vbComp As Object
    Dim dossier As String
    Dim fso As Object
    Dim nomFichier As String
    Dim codeTexte As String
    Dim cheminTempFrm As String
    Dim nomComp As String
    
    If ThisWorkbook.Path = "" Then
        MsgBox "Enregistre d'abord le classeur avant l'export.", vbExclamation
        Exit Sub
    End If
    
    dossier = ThisWorkbook.Path & "\Export_TXT"
    
    Set fso = CreateObject("Scripting.FileSystemObject")
    If Not fso.FolderExists(dossier) Then
        fso.CreateFolder dossier
    End If
    
    For Each vbComp In ThisWorkbook.VBProject.VBComponents
        
        nomComp = vbComp.Name
        
        Select Case vbComp.Type
        
            Case 1, 2
                nomFichier = dossier & "\" & nomComp & ".txt"
                EcrireTexteDansFichier nomFichier, LireCodeComponent(vbComp, nomComp)
            
            Case 3
                cheminTempFrm = dossier & "\" & nomComp & ".frm"
                
                On Error Resume Next
                Kill cheminTempFrm
                Kill dossier & "\" & nomComp & ".frx"
                Kill dossier & "\" & nomComp & ".txt"
                On Error GoTo 0
                
                vbComp.Export cheminTempFrm
                
                nomFichier = dossier & "\" & nomComp & ".txt"
                codeTexte = LireFichierTexte(cheminTempFrm)
                EcrireTexteDansFichier nomFichier, codeTexte
                
                On Error Resume Next
                Kill cheminTempFrm
                Kill dossier & "\" & nomComp & ".frx"
                On Error GoTo 0
            
            Case 100
                If nomComp = NOM_THISWORKBOOK Or nomComp = NOM_FEUILLE_CIBLE Then
                    nomFichier = dossier & "\" & nomComp & ".txt"
                    EcrireTexteDansFichier nomFichier, LireCodeComponent(vbComp, nomComp)
                End If
        
        End Select
    Next vbComp
    
    MsgBox "Export TXT terminé dans : " & dossier, vbInformation

End Sub

Private Function LireCodeComponent(ByVal vbComp As Object, ByVal nomComp As String) As String

    Dim codeTexte As String
    
    If vbComp.CodeModule.CountOfLines > 0 Then
        codeTexte = vbComp.CodeModule.Lines(1, vbComp.CodeModule.CountOfLines)
    Else
        codeTexte = ""
    End If
    
    LireCodeComponent = "Attribute VB_Name = """ & nomComp & """" & vbCrLf & codeTexte

End Function

Private Sub EcrireTexteDansFichier(ByVal chemin As String, ByVal contenu As String)

    Dim numFichier As Integer
    
    On Error Resume Next
    Kill chemin
    On Error GoTo 0
    
    numFichier = FreeFile
    Open chemin For Output As #numFichier
    Print #numFichier, contenu
    Close #numFichier

End Sub

Private Function LireFichierTexte(ByVal chemin As String) As String

    Dim fso As Object
    Dim ts As Object
    
    Set fso = CreateObject("Scripting.FileSystemObject")
    Set ts = fso.OpenTextFile(chemin, 1, False, -1)
    LireFichierTexte = ts.ReadAll
    ts.Close

End Function
